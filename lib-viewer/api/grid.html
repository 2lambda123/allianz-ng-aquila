import { NxGridModule } from '@aposin/ng-aquila/grid';

Directives

NxColComponent

Selector:

[nxCol]

Properties
Name Description
@Input('nxCol')

col: string

Number of columns used.

Values: 1 - 12, default value: 12.
@Input('nxAlignSelf')

itemSelf: string

The alignment for a column inside the flexible container.
@Input('nxColOffset')

offset: string

The number of columns the column should be offset.

Values: 1 - 12, default value: 12.
@Input('nxColOrder')

order: string

Order of the column within the row.

NxLayoutComponent

Selector:

[nxLayout]

Properties
Name Description
@Input('nxLayout')

classNames: string

Type of layout.

Values: grid | grid nogutters | grid maxwidth | grid nopadding. Default value: grid.

NxRowComponent

Selector:

[nxRow]

Properties
Name Description
@Input()

nxRow: string

Values: row | row-reverse

Default value: row
@Input('nxRowAlignContent')

nxRowAlignContent: string

Similar to nxRowAlignItems, but instead of aligning flex items, it aligns flex lines.
@Input('nxRowAlignItems')

nxRowAlignItems: string

The default alignment for items inside the flexible container.
@Input('nxRowJustify')

nxRowJustify: string

Align items on the main axis (horizontally).
@Input('nxRowWrap')

nxRowWrap: RowWrapping

How the flexible items should be wrapped.

Additional type aliases

ColSelfAlignment

Type for the available alignment values of a column inside the flexible container.

Type: 'auto' | 'start' | 'end' | 'center' | 'baseline' | 'stretch'

ColOrder

Type for the available values for setting the order of a column within a row.

Type: 'first' | 'last' | 'unordered'

RowJustification

Type: 'start' | 'end' | 'center' | 'between' | 'around'

RowContentAlignment

Type: 'start' | 'end' | 'center' | 'between' | 'around' | 'stretch'

RowItemsAlignment

Type: 'start' | 'end' | 'center' | 'between' | 'stretch'

RowWrapping

Type: 'wrap' | 'nowrap' | 'reverse'
